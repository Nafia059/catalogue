/* ---------- flipbook modal logic (self-contained, no external library) ---------- */
const backdrop = document.getElementById('modalBackdrop');
const flipEl = document.getElementById('flipbook');
const modalTitle = document.getElementById('modalTitle');
const downloadBtn = document.getElementById('downloadBtn');
const pageIndicator = document.getElementById('pageIndicator');

let activePages = [];   // DOM elements for the currently open catalogue
let currentIndex = 0;
let isAnimating = false;

function openCatalogue(i){
  const cat = CATEGORIES[i];
  modalTitle.textContent = cat.name + ' — Catalogue';
  downloadBtn.href = cat.pdfUrl;

  flipEl.innerHTML = '';
  activePages = cat.pages.map((html, idx) => {
    const page = document.createElement('div');
    page.className = 'page' + (idx === 0 ? ' active' : '');
    page.innerHTML = html;
    flipEl.appendChild(page);
    return page;
  });
  currentIndex = 0;
  pageIndicator.textContent = `Page 1 / ${activePages.length}`;

  backdrop.classList.add('open');
}

  const next = activePages[newIndex];

  current.classList.remove('active');
  current.classList.add(direction === 'next' ? 'exit-next' : 'exit-prev');

  next.classList.add(direction === 'next' ? 'enter-from-next' : 'enter-from-prev');
  void next.offsetWidth; // force reflow so the entering transform applies first

  requestAnimationFrame(() => {
    next.classList.remove('enter-from-next', 'enter-from-prev');
    next.classList.add('active');
  });

  setTimeout(() => {
    current.classList.remove('exit-next', 'exit-prev');
    currentIndex = newIndex;
    pageIndicator.textContent = `Page ${currentIndex + 1} / ${activePages.length}`;
    isAnimating = false;
  }, 460);
}

document.getElementById('nextPage').addEventListener('click', () => goToPage(currentIndex + 1, 'next'));
document.getElementById('prevPage').addEventListener('click', () => goToPage(currentIndex - 1, 'prev'));

document.getElementById('closeModal').addEventListener('click', () => {
  backdrop.classList.remove('open');
});
backdrop.addEventListener('click', (e) => {
  if (e.target === backdrop) backdrop.classList.remove('open');
});
document.addEventListener('keydown', (e) => {
  if (!backdrop.classList.contains('open')) return;
  if (e.key === 'ArrowRight') goToPage(currentIndex + 1, 'next');
  if (e.key === 'ArrowLeft') goToPage(currentIndex - 1, 'prev');
  if (e.key === 'Escape') backdrop.classList.remove('open');
});
